++i 와 i++

// prefix increment operator++

	INT& operator++( )
	{
		++val;
		return *this;		// this포인터 - 자기자신에 *면 진짜 자기자신
	}

// posfix increment operator++
	int operator++( int )
	{
		// 내 현재 값을 저장한다
		int temp { val };	// INT temp { val }
		// 내 값이 1 증가 한다
		++( *this );
		// 저장한 값을 되돌려 준다
		return temp;
	}
