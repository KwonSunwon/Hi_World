#include <iostream>
#include <string>
#include "save.h"

using namespace std;

//  Animal	( int age, string name ), move()
//	  /\
//	 /  \
// Dog  Bird
//(달리기) (날기)

class Animal {
protected:
	int age;
	string name;
public:
};

class Dog : public Animal {
public:
	Dog ( string s, int n ) {
		name = s;
		age = n;
	}

	void move ( ) {
		cout << name << " 달린다" <<endl;
	}
};

class Bird : public Animal {
public:
	Bird ( string s, int n ) {
		name = s;
		age = n;
	}

	void move ( ) {
		cout << name << " 난다" << endl;
	}
};

int main()
{
	Dog d {"진돌이", 3};
	Bird b {"양진이", 5};

	d.move();	// 진돌이 달린다
	b.move();	// 양진이 난다

	save ("소스.cpp");
}
