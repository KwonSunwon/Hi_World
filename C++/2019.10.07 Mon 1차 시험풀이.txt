//
// 시험 1 답지 확인

#include <iostream>
#include <fstream>
#include <random>
#include "save.h"
using namespace std;

int main()
{
	int a[] { 2, -8, 5, -1, 10, -7, 9, -3, 4, -6 };

	sort( begin( a ), end( a ), []( int a, int b ) {
		return abs( a ) < abs( b );
	});

	for( int d : a )
		cout << d << ' ';
	cout << endl;

	save( "소스.cpp" );
}

//  - 1, 2, 4, 8 보다 큰 메모리가 필요해서
//  - POD 가지고 프로그램하기엔 규모도 크고 이해가 어렵고 등등...
//
// 시험 1 답지 확인
// 과제
//
// 10. 09 수 34교시 정상 수업합니다.
//-----------------------------------------------------------------------------

#include <iostream>
#include <fstream>
#include <random>
#include "save.h"
using namespace std;

void makeData();
string fileName { "정수2억개.txt" };

default_random_engine dre;
uniform_int_distribution<int> uid;

int main()
{
	makeData();

	int data, max { 0 };
	ifstream in { fileName };

	for( int i = 0; i < 2'0000'0000; ++i )
	{
		in >> data;
		if( max < data )
			max = data;
	}

	cout << "가장 큰 정수 : " << max << endl;

	save( "소스.cpp" );
}

void makeData()	//  파일을 다 만들면 그 후 삭제해야됨
{
	ofstream out { fileName };

	for( int i = 0; i < 2'0000'0000; ++i )
	{
		if( ( i % 1000'0000 ) == 0 )
			cout << "기록 중 " << i << endl;
		out << uid( dre );
	}
}
